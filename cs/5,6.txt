def encrypt(text, key):
    if key <= 1:
        return text

    rail = ['' for _ in range(key)]
    row = 0
    direction = 1

    for char in text:
        rail[row] += char
        row += direction

        if row == 0 or row == key - 1:
            direction *= -1

    return ''.join(rail)


def decrypt(cipher, key):
    if key <= 1:
        return cipher

    pattern = []
    row = 0
    direction = 1

    for _ in cipher:
        pattern.append(row)
        row += direction
        if row == 0 or row == key - 1:
            direction *= -1

    rail_lengths = [pattern.count(i) for i in range(key)]

    rails = []
    index = 0
    for length in rail_lengths:
        rails.append(cipher[index:index + length])
        index += length

    result = []
    rail_index = [0] * key

    for r in pattern:
        result.append(rails[r][rail_index[r]])
        rail_index[r] += 1

    return ''.join(result)

import math
import os

def encryptMessage(msg, key):
    cipher = ""
    col = len(key)
    msg_len = float(len(msg))
    row = int(math.ceil(msg_len / col))

    fill_null = int((row * col) - msg_len)
    msg += '_' * fill_null

    matrix = [msg[i: i + col] for i in range(0, len(msg), col)]

    key_order = sorted([(char, i) for i, char in enumerate(key)])

    for char, curr_idx in key_order:
        cipher += ''.join([matrix[r][curr_idx] for r in range(row)])

    return cipher


def decryptMessage(cipher, key):
    col = len(key)
    msg_len = float(len(cipher))
    row = int(math.ceil(msg_len / col))

    key_order = sorted([(char, i) for i, char in enumerate(key)])

    dec_cipher = [[''] * col for _ in range(row)]
    msg_indx = 0

    for char, curr_idx in key_order:
        for j in range(row):
            dec_cipher[j][curr_idx] = cipher[msg_indx]
            msg_indx += 1

    msg = ''.join(sum(dec_cipher, []))
    return msg.rstrip('_')
